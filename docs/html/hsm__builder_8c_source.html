229 static hsm_state RunGenericEvent( hsm_status status )
230 {
231  hsm_state next_state=NULL;
232  state_t* state= StateFromStatus( status );
233  const process_t* et;
234  // look through the specified event processors
235  for (et= state->process; et && !next_state; et=et->next) {
236  if (et->flags & ProcessCallback) {
237  next_state= CALL_PROCESS( et, status );
238  }
239  // this block of code could have been installed as a custom process callback
240  // but this avoids a separate function call, and is pretty straight forward
241  // dont know, maybe i will revist for improved code aethetics
242  else {
243  handler_t * handler= (handler_t*)et;
244  // determine if some guard blocks the event from running
245  guard_t* guard;
246  for (guard= handler->guard; guard; guard=guard->next) {
247  if (!CALL_GUARD( guard, status )) {
248  break;
249  }
250  }
251  // no guard blocks this handler from running,:
252  if (!guard) {
253  // run action(s)
254  action_t* at;
255  for (at= handler->actions; at; at=at->next) {
256  at->run( status, at->action_data );
257  }
258  // transition to target, or flag as handled.
259  if (handler->target) {
260  next_state= (hsm_state) handler->target->clientData;
261  }
262  else {
263  next_state= HsmStateHandled();
264  }
265  }
266  }
267  }
268  return next_state;
269 }

697 {
698  hsm_state ret= NULL;
699  builder_t * builder= ((builder_t*)status->ctx);
700  state_t* current= Builder_CurrentState( builder );
701  switch (status->evt->type)
702  {

715  }

717  case _hsm_enter_raw: {
718  const RawEnterEvent* event= (const RawEnterEvent*)status->evt;
719  if (!current->desc.enter && event->enter) {
720  current->desc.enter= event->enter;
721  ret= BuildingBody();
722  }
723  else {
724  Builder_Error( builder, current->desc.enter ? "enter already specified." : "enter is null." );
725  ret= HsmStateError();
726  }
727  }
728  break;
729  case _hsm_exit_raw: {
730  const RawActionEvent* event= (const RawActionEvent*)status->evt;
731  if (!current->desc.exit && event->action) {
732  current->desc.exit= event->action;
733  ret= BuildingBody();
734  }
735  else {
736  Builder_Error( builder, current->desc.exit ? "exit already specified." : "exit is null." );
737  ret= HsmStateError();
738  }
739  }
740  break;
741  case _hsm_exit_ud: {
742  const ActionEvent* event= (const ActionEvent*)status->evt;
743  if (!current->desc.exit && event->action) {
744  current->desc.exit= RunGenericExit;
745  current->exit= event->action;
746  current->exit_ud= event->action_data;
747  ret= BuildingBody();
748  }
749  else {
750  Builder_Error( builder, current->desc.exit ? "exit already specified." : "exit is null." );
751  ret= HsmStateError();
752  }
753  }
754  break;
755  case _hsm_process_raw: {
756  const RawProcessEvent* event= (const RawProcessEvent*)status->evt;
757  if (event->process) {
758  NewProcessRaw( current, event->process );
759  ret= BuildingBody();
760  }
761  else {
762  Builder_Error( builder, "process is null." );
763  ret= HsmStateError();
764  }
765  }
766  break;
767  case _hsm_process_ud: {
768  const ProcessEventUd* event= (const ProcessEventUd*)status->evt;
769  if (event->process) {
770  NewProcessUD( current, event->process, event->process_data );
771  ret= BuildingBody();
772  }
773  else {
774  Builder_Error( builder, "process is null." );
775  ret= HsmStateError();
776  }
777  }
778  break;
779  // start building a event handler
780  case _hsm_guard_raw:
781  case _hsm_guard_ud:
782  {
783  ret= BuildingHandler();
784  }
785  break;
786  case _hsm_end: {
787  // setup the event handler, this also a key that the state is good to go.
788  current->desc.process= RunGenericEvent;
789 
790  // if this was the last matching begin/end pair, we're done
791  // ( we dont use !current, b/c of potential containment in external states )
792  if (--builder->count==0) {
793  builder->current= 0;
794  ret= Building();
795  }
796  else {
797  builder->current= (state_t*) current->desc.parent;
798  ret= BuildingBody();
799  }
800  }
801  break;
802  }
803  return ret;
804 }

854  switch (status->evt->type) {
855  case _hsm_goto: {
856  const StateEvent* event= (const StateEvent*)status->evt;
857  const int go= event->id;
858  hash_entry_t* target= Hash_FindEntry( &(builder->hash), go );
859  if (!handler->target && target) {
860  handler->target= target;
861  ret= HsmStateHandled();
862  }
863  else {
864  Builder_Error( builder, handler->target ? "goto already specified" : "unknown goto target" );
865  ret= HsmStateError();
866  }
867  }
868  break;
869  case _hsm_guard_raw:
870  case _hsm_guard_ud:
871  {
872  const GuardEvent* guard_event= (const GuardEvent*)status->evt;
873  // on append, we want to add this guard to us
874  // otherwise we let the parent state start a new event handler
875  if (guard_event->append) {
876  guard_t* guard= NewGuardFromEvent( status, handler );
877  if (guard) {
878  ret= HsmStateHandled();
879  }
880  else {
881  Builder_Error( builder, "couldnt create guard");
882  ret= HsmStateError();
883  }
884  }
885  }
886  break;
887  case _hsm_action_ud: {
888  const ActionEvent* action_event= (const ActionEvent*)status->evt;
889  if (NewAction( handler, action_event->action, action_event->action_data )) {
890  ret= HsmStateHandled();
891  }
892  else {
893  Builder_Error( builder, !action_event->action ? "no action specified" : "couldnt allocate action");
894  ret= HsmStateError();
895  }
896  }
897  break;
898  };
899  }
900  return ret;
901 }
