162 hsm_bool HsmStart( hsm_machine hsm, hsm_state first_state )
163 {
164  assert( hsm );
165  assert( first_state && "expected valid first state for init" );
166  assert( (!hsm || !hsm->current) && "already ran init" );
167 
168  if (hsm && !hsm->current && first_state )
169  {
170  // the specified starting state isn't necessarily the *top* state:
171  // we need to walk up to the top state, then walk down to and including our first state
172  HsmRecursiveEnter( hsm, first_state, NULL );
173 
174  // statecharts run enter *then* init
175  // ( note: init can move us into a new state )
176  HsmInit( hsm, NULL );
177  }
178 
179  return HsmIsRunning( hsm );

183 hsm_bool HsmSignalEvent( hsm_machine hsm, hsm_event evt )
184 {
185  hsm_bool okay= HSM_FALSE;
186  if (hsm && hsm->current)
187  {
188  // bubble the event up the hierarchy until we get a valid respose
189  // ( or until we run off the top of the tree. )
190  hsm_state next_state= NULL;
191  hsm_state handler= hsm->current;
192  hsm_context_iterator_t it;
193  HsmContextIterator( &it, HSM_STACK( hsm ) );
194  do {
195  if (handler->process) {
196  hsm_status_t status= { hsm, handler, it.context, evt };
197  next_state= handler->process( &status ) ;
198  if (next_state) {
199  break;
200  }
201  }
202  HsmParentContext( &it );
203  handler= handler->parent;
204  }
205  while (handler);
206 
207  // handlers are supposed to return HsmStateHandled
208  if (!next_state) {
209  if (hsm_global_callbacks.on_unhandled_event) {
210  hsm_status_t status= { hsm, NULL, NULL, evt };
211  hsm_global_callbacks.on_unhandled_event( &status, hsm_global_callbacks.user_data );
212  }
213  }
214  else
215  if (next_state== HsmStateHandled()) {
216  okay= HSM_TRUE;
217  }
218  else
219  if (next_state== HsmStateFinal() || next_state==HsmStateError()) {
220  // we say this is okay, in the sense that something happened.
221  // trigger a callback to inform the user?
222  if (hsm->current != next_state) {
223  hsm->current = next_state;
224  okay= HSM_TRUE;
225  }
226  }
227  else {
228  // transition, and if all is well, init
229  if (!HsmTransition( hsm, handler, next_state, evt )) {
230  hsm->current= HsmStateError();
231  }
232  else {
233  HsmInit( hsm, evt );
234  }
235  }
236  }
237  return okay;
238 }

241 // warning: this indirectly alters the current state and context stack ( via HsmEnter )
242 static void HsmInit( hsm_machine hsm, hsm_event cause )
243 {
244  hsm_state initial_state;
245  while ( initial_state= hsm->current->initial )
246  {
247  hsm_bool init_moves_to_child= initial_state->parent == hsm->current;
248  assert( init_moves_to_child && "malformed statechart: init doesnt move to child state" );
249 
250  if (hsm_global_callbacks.on_init) {
251  hsm_context_stack_t* stack= HSM_STACK( hsm );
252  hsm_status_t status= { hsm, hsm->current, stack ? stack->context: 0, cause };
253  hsm_global_callbacks.on_init( &status, hsm_global_callbacks.user_data );
254  }
255 
256  if (!init_moves_to_child) {
257  hsm->current= HsmStateError();
258  break;
259  }
260  else {
261  // continue the enter=>init pattern
262  // ( the spec says enter runs before init. it sounds insane, but works out well. )
263  HsmEnter( hsm, initial_state, cause );
264  }
265  }
266 }

269 // warning: this directly alters the current state ( and also modifies the context stack. )
270 // the original code did not, but this change help simplify: init, exit, and transition.
271 static void HsmEnter( hsm_machine hsm, hsm_state state, hsm_event cause )
272 {
273  //FIXME-stravis: handle invalid states better?
274  const hsm_bool valid_state= (state && state->depth >=0);
275  assert( valid_state );
276  if (valid_state) {
277  // note: each state gets the context of its parent in entry
278  // and can optionally generate a new context in turn
279  hsm_context_stack_t* stack= HSM_STACK( hsm );
280  hsm_status_t status= { hsm, state, stack ? stack->context: 0, cause };
281 
282  if (state->enter) {
283  status.ctx= state->enter( &status );
284  }
285 
286  // push the new context, the stack handles dupes.
287  HsmContextPush( stack, status.ctx );
288  hsm->current= state;
289 
290  // informational callback, passing in new context
291  if (hsm_global_callbacks.on_entered) {
292  hsm_global_callbacks.on_entered( &status, hsm_global_callbacks.user_data );
293  }
294  }
295 }

300 void HsmExit( hsm_machine hsm, hsm_event cause )
301 {
302  // note: exit, just like process, gets the context enter created
303  hsm_context popped;
304  hsm_state state= hsm->current;
305  hsm_context_stack_t * stack= HSM_STACK( hsm );
306  hsm_status_t status= { hsm, state, stack ? stack->context: 0, cause };
307 
308  // informational callback, passing the old context
309  if (hsm_global_callbacks.on_exiting) {
310  hsm_global_callbacks.on_exiting( &status, hsm_global_callbacks.user_data );
311  }
312 
313  if (state->exit) {
314  state->exit( &status );
315  }
316 
317  // exit pops the context that enter had created.
318  hsm->current= state->parent;
319  popped= HsmContextPop( stack );
320 
321  // finally: let the user know
322  if (popped) {
323  if (hsm_global_callbacks.on_context_popped) {
324  hsm_global_callbacks.on_context_popped( &status, hsm_global_callbacks.user_data );
325  }
326  // call the "hsm_callback_context_popped" of the item thats being popped
327  if (popped->popped) {
328  popped->popped( popped );
329  }
330  }
331 }

369 static hsm_bool HsmTransition( hsm_machine hsm, hsm_state source, hsm_state target, hsm_event cause )
370 {
371  // above ( I. ) specifically designates the lca of 'source' and 'target'.
372  // which can only, then, be the same node as 'source' or, more likely, a node less deep than 'source'.
373  // since 'source' is the node that handled the event, and since events propogate from 'current' up to the top of the tree
374  // that also means 'source' is either the same node as, or higher than, 'current'.
375  // ie. lca->depth <= source->depth <= current->depth
376  // therefore: as a first step, bring 'current' to 'source', and work towards lca from there.
377  while( hsm->current != source ) {
378  HsmExit( hsm, cause );
379  ERROR_IF_NULL( hsm->current, "jumped past top" );
380  }
381 
382  // quick check for self transition: the source targeted itself ( III. above )
383  if ( hsm->current == target ) {
384  HsmExit( hsm, cause );
385  // ( <--- note: in uml transitions actions would take place here )
386  // ( in hsm_statechart, they've already happened by now )
387  HsmEnter( hsm, target, cause );
388  }
389  else {
390  // the easiest entry path record is an object on the stack...
391  // doesnt worry too much about stack overflow,
392  // instead it lets the user control it via HSM_MAX_DEPTH...
393  int pt=0;
394  hsm_state track= target;
395  hsm_state* path_to_target=(hsm_state*) alloca( target->depth * sizeof(hsm_state) );
396  hsm_bool external_transition= 0;
397  ERROR_IF_NULL( path_to_target, "out of space" );
398 
399  // source deep than target?
400  if (hsm->current->depth > track->depth) {
401  // *exit* up to the same level
402  while( hsm->current->depth > track->depth) {
403  HsmExit( hsm, cause ); // we exit, then move, b/c, if we don't leave the node we don't want an exit
404  ERROR_IF_NULL( hsm->current, "jumped past top" );
405  }
406  }
407  // target deeper than source?
408  else {
409  // *record* its path up to the same level
410  while (track->depth > hsm->current->depth ) {
411  path_to_target[pt++]= track;
412  track= track->parent;
413  ERROR_IF_NULL( track, "jumped past top" );
414  }
415 
416  // trigger an external transition? (re: II. above)
417  // by design, in hsm-statechart all transitions are internal.
418  #ifdef HSM_USE_EXTERNAL_TRANSITIONS
419  // since track has now risen along the path of target-to-root
420  // and has reached the the level of source....
421  // if they are the same node, then source was an ancestor of target.
422  if (hsm->current == track) {
423  HsmExit( hsm, cause ); // this bumps the current state up
424  path_to_target[pt++]= track; // record that we need to renter it
425  track= hsm->current; // and bump track up to the same level
426  }
427  #endif
428  }
429 
430  // keep going up together till current and track have found each other
431  // ( keep exiting 'current' as it goes up; keep recording 'track' as *it* goes up )
432  while (hsm->current!= track) {
433  HsmExit( hsm, cause );
434  path_to_target[pt++]= track;
435  track= track->parent;
436  ERROR_IF_NULL( hsm->current&&track, "jumped past top" );
437  }
438 
439  // ( <--- note: in uml transitions actions would take place here )
440  // ( in hsm_statechart, they've already happened by now )
441 
442  // now have current use path the we just recorded:
443  // it's turtles all the way down.
444  while (pt>0) {
445  track= path_to_target[--pt];
446  HsmEnter( hsm, track, cause );
447  }
448  }
449  // note: on error we will have already returned HSM_FALSE
450  return HSM_TRUE;
451 }
